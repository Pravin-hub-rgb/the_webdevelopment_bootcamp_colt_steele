Loops

While Loops

While loops continue running as long as the test condition is true.

                        

                            let num = 0;
                            while (num < 5) {
                             console.log(num);
                              num++;
                            }
                        
                
                    

                        const secret = 'BabyYoda';
                        let guess = prompt("Enter the secret code");
                        while(guess!==secret){
                          guess = prompt("Enter the secret code");
                        }
                        console.log("Congratulations you got the secret!")

The Break Keyword

This is mostly used inside while loop

→ This let the program to exit out of loop immediately

                        

                            let targetNum = Math.floor(Math.random()*10);
                            let guess;
                            
                            while(true){
                              guess = Math.floor(Math.random()*10);
                              if(guess === targetNum){
                               console.log(`Correct! Guessed: ${guess} & target was: ${targetNum}`);
                               break;
                              }else{
                               console.log(`Guessed ${guess}....Incorrect`);
                            }
                            }

Another Example (will keep on looping until you type 'stop')

                            

                                let msg = prompt("Enter your message!");
                                while(true){
                                  msg=prompt(msg);
                                  if(msg==="stop") break;
                                }
                                console.log("The message has been stopped!")

Guessing Game Code

                    

                        let maximum = parseInt(prompt("Enter the maximum number"));
                        while (!maximum) {
                            maximum = parseInt(prompt("Enter a valid number"));
                        }
                        const target = Math.floor(Math.random() * maximum) + 1;
                        
                        let guess = prompt("Enter your first guess!");
                        let attempts = 1;
                        while (parseInt(guess) !== target) {
                            attempts++;
                            if (guess === 'q') break;
                            if (guess > target) {
                                guess = prompt("Too high! Enter a new guess:");
                            } else {
                                guess = prompt("Too low! Enter a new guess:");
                            }
                        }
                        if (guess === 'q') {
                            console.log("ok, you quit!");
                        } else {
                            console.log(`you got it! It took you ${attempts}`)
                        }

For...of (Mdn)

A nice and easy way of iterating over arrays (or other iterable objects)

Syntax

                        

                            for (variable of interable) {
                                statement
                              }

Example

                            

                                const subreddits = ['cringe','books','chickens','funny','pics','soccer','dance'];
                                for (let sub of subreddits){
                                  console.log(sub);
                                }
                            
                        
                            

                                const seatingChart = [
                                ['Kristen','Erik','Namita'],
                                ['Geoffrey','Juanita','Antonio','Kevin'],
                                ['Yuma','Sakura','Jack','Erika']
                                ]
                                for(let row of seatingChart){
                                  for(let student of row){
                                  console.log(student)
                                }
                                }

We can use for...of loop for strings too

                            

                                for(let char of "hello world"){
                                    console.log(char);
                                  }

Interating Over Objects

Objects are not interable. So, we can not use for...of loop directly.

for...in loop

It can be used in objects to iterate over them

                            

                                let testScores ={
                                    keenan:80,
                                    damon:67,
                                    kim:89,
                                    shawn:91,
                                    marlon:72,
                                    dwayne:77,
                                    nadia:83,
                                    elvira:97,
                                    diedre:81,
                                   vonnie:60
                                  }
                                  for(let person in testScores){
                                    console.log(`${person} scored ${testScores[person]}`)
                                  }

Special Objects Methods

Object.keys(testScores)

Returns an array made out of all the keys in an objects

                           
                           
                            ["keenan", "damon", "kim", "shawn", "marlon", "dwayne", "nadia", "elvira", "diedre", "vonnie"]

Object.values(testScores)

Returns an array made out of all the values in an objects

                           
                           
                            [80, 67, 89, 91, 72, 77, 83, 97, 81, 60]

Object.entries(testScores)

Returns an array made of all the key value pairs in an objects

                           
                           
                            (10) [Array(2), Array(2), Array(2), Array(2), Array(2), Array(2), Array(2), Array(2), Array(2), Array(2)]

                            4: (2) ["marlon", 72]

Now using special objects methods we can use for...of loop

                            

                                //adding all the score
                                let totalSum=0;
                                for(let score of Object.values(testScores)){
                                  totalSum+=score;
                                }
                                console.log(totalSum)
